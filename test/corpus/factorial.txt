/// Computes the factorial of `n`.
///
/// - Requires: `n` must be greater or equal to 0.
fun factorial(_ n: Int) -> Int {
  if n < 2 { 1 } else { n * factorial(n - 1) }
}

public fun main() {
  let _ = factorial(6)
}
